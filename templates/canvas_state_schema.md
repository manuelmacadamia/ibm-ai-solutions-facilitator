# Canvas state schema

Usa questa struttura come memoria di lavoro interna.

```yaml
project_brief:
  user:
  current_situation:
  problem:
  desired_outcome:
  sponsor_decision_maker:
  constraints:
  relational_context:
    protect_or_grow:
    should_not_lose:
    people_time_or_quality_to_recover:

team_map:
  core_team: []
  experts: []
  stakeholders: []

assumptions_questions:
  high_risk_high_uncertainty: []

as_is_scenario_map:
  phases: []
  prioritized_pain_points: []

intent_canvas:
  core_ai_intents: []
  user_benefit:
  business_benefit:
  why_ai:
  intent_statement:
  doubts_to_validate:

big_idea_vignettes:
  ideas: []
  selected_big_idea:
  parked_alternatives:

data_inventory:
  public_data:
    have:
    need_want:
    nice_to_have:
  private_data:
    have:
    need_want:
    nice_to_have:
  user_data:
    have:
    need_want:
    nice_to_have:
  critical_data_sources: []
  minimum_data_for_test:
  data_gaps:

understanding_map:
  critical_sources: []

reasoning_statement:
  formula:
  original_big_idea:
  refined_big_idea:
  open_assumptions:
  first_validation:

effects_safeguards:
  primary_effects:
  secondary_effects:
  tertiary_effects:
  risks: []

storyboard_test_plan:
  moments: []
  hypothesis:
  method:
  participants:
  go_no_go_criterion:

playback:
  user_and_pain_point:
  intent_and_big_idea:
  data_and_reasoning:
  risks_and_safeguards:
  next_experiment:
  decision:
  owner:
  date:
  verification_criterion:
```
